  <script>

    const DEFAULT_TIMER_MS = 30 * 60 * 1000; // 30 minutes in milliseconds

    let localTimerInterval = null;

    // State Management

    let state = {

      tasks: [],

      savedTasks: [],

      currentDate: "",

      currentPickId: null,

      selectedPriority: 'medium',

      completionHistory: {},

      audioEnabled: true,

      streak: 0,

      timer: {

        activeTaskId: null,

        isRunning: false,

        targetTime: null,

        remainingTime: DEFAULT_TIMER_MS

      }

    };

    // Date formatting helper (YYYY-MM-DD)

    function formatDate(date) {

      const year = date.getFullYear();

      const month = String(date.getMonth() + 1).padStart(2, '0');

      const day = String(date.getDate()).padStart(2, '0');

      return `${year}-${month}-${day}`;

    }

    // Load State from LocalStorage

    function loadState() {

      const saved = localStorage.getItem('pickflow_state_data_05');

      const todayStr = formatDate(new Date());

      if (saved) {

        try {

          state = JSON.parse(saved);

          if (!state.tasks) state.tasks = [];

          if (!state.savedTasks) state.savedTasks = [];

          if (!state.completionHistory) state.completionHistory = {};

          if (state.audioEnabled === undefined) state.audioEnabled = true;

          if (state.streak === undefined) state.streak = 0;

          if (!state.timer) {

            state.timer = { activeTaskId: null, isRunning: false, targetTime: null, remainingTime: DEFAULT_TIMER_MS };

          }

         

          if (state.currentDate !== todayStr) {

            state.tasks = [];

            state.currentDate = todayStr;

          }

        } catch (e) {

          console.error("Failed to parse saved state", e);

        }

      } else {

        state.tasks = [

          { id: crypto.randomUUID(), text: "📝 Draft email updates for the clients", priority: 'low', completed: false, createdAt: new Date() },

          { id: crypto.randomUUID(), text: "🏋️ Go to gym and run for 30 minutes", priority: 'high', completed: false, createdAt: new Date() },

          { id: crypto.randomUUID(), text: "💻 Refactor Picker app UI styling", priority: 'medium', completed: false, createdAt: new Date() }

        ];

        state.currentDate = todayStr;

      }

      // Check for Background Timer Progress Upon Loading

      handleBackgroundTimerRestoration();

    }

    function saveState() {

      localStorage.setItem('pickflow_state_data_05', JSON.stringify(state));

    }

    // --- Timer Logic (Background & UI) ---

   

    function formatTime(ms) {

      if (ms <= 0) return "00:00";

      const totalSeconds = Math.ceil(ms / 1000);

      const m = Math.floor(totalSeconds / 60);

      const s = totalSeconds % 60;

      return `${String(m).padStart(2, '0')}:${String(s).padStart(2, '0')}`;

    }

    function updateTimerUI(ms) {

      document.getElementById('timerDisplay').textContent = formatTime(ms);

      const icon = document.getElementById('timerIcon');

      const btn = document.getElementById('timerToggleBtn');

      if (state.timer.isRunning) {

        icon.className = "fa-solid fa-pause text-xl";

        icon.classList.remove('ml-0.5'); // Remove play icon offset

        btn.className = "w-12 h-12 rounded-xl bg-amber-500 hover:bg-amber-400 text-slate-900 flex items-center justify-center shadow-lg shadow-amber-500/20 transition active:scale-95";

      } else {

        icon.className = "fa-solid fa-play text-xl ml-0.5";

        btn.className = "w-12 h-12 rounded-xl bg-blue-600 hover:bg-blue-500 text-white flex items-center justify-center shadow-lg shadow-blue-500/20 transition active:scale-95";

      }

    }

    function toggleTimer() {

      if (state.timer.isRunning) {

        // Pause it

        state.timer.isRunning = false;

        state.timer.remainingTime = state.timer.targetTime - Date.now();

        state.timer.targetTime = null;

        clearInterval(localTimerInterval);

        playSynthSound('click');

      } else {

        // Start / Resume it

        state.timer.isRunning = true;

        state.timer.targetTime = Date.now() + state.timer.remainingTime;

        playSynthSound('tick');

        startTimerLoop();

      }

      updateTimerUI(state.timer.remainingTime);

      saveState();

    }

    function startTimerLoop() {

      if (localTimerInterval) clearInterval(localTimerInterval);

     

      localTimerInterval = setInterval(() => {

        const left = state.timer.targetTime - Date.now();

        if (left <= 0) {

          // Timer finished naturally!

          clearInterval(localTimerInterval);

          handleTimerComplete();

        } else {

          updateTimerUI(left);

        }

      }, 200);

    }

    function handleTimerComplete() {

      updateTimerUI(0);

     

      const taskId = state.timer.activeTaskId;

      clearActiveTimer();

     

      // Complete the task automatically

      if (taskId) {

        markTaskComplete(taskId, true);

        showFeedbackToast("Time is up! Task auto-completed.");

      }

     

      // Close the modal if it's open

      closeCongratsModal();

    }

    function clearActiveTimer() {

      if (localTimerInterval) clearInterval(localTimerInterval);

      state.timer = {

        activeTaskId: null,

        isRunning: false,

        targetTime: null,

        remainingTime: DEFAULT_TIMER_MS

      };

      saveState();

    }

    // Handles the timer if the user closed the app and came back

    function handleBackgroundTimerRestoration() {

      const t = state.timer;

      if (!t || !t.activeTaskId) return;

      const taskExists = state.tasks.find(x => x.id === t.activeTaskId && !x.completed);

     

      if (!taskExists) {

        // Task was deleted or already completed elsewhere, clean up

        clearActiveTimer();

        return;

      }

      if (t.isRunning && t.targetTime) {

        const left = t.targetTime - Date.now();

        if (left <= 0) {

          // User was away and time expired

          markTaskComplete(t.activeTaskId, true);

          showFeedbackToast(`Welcome back! Your task "${taskExists.text}" was auto-completed while you were away.`);

          clearActiveTimer();

        } else {

          // Still running, reopen the modal

          state.currentPickId = t.activeTaskId;

          populateModalDetails(taskExists);

          updateTimerUI(left);

          startTimerLoop();

          openCongratsModal();

        }

      } else if (!t.isRunning && t.remainingTime > 0 && t.remainingTime < DEFAULT_TIMER_MS) {

        // It was paused midway. Re-open modal

        state.currentPickId = t.activeTaskId;

        populateModalDetails(taskExists);

        updateTimerUI(t.remainingTime);

        openCongratsModal();

      }

    }

    function markTaskComplete(id, auto = false) {

      const task = state.tasks.find(t => t.id === id);

      if (!task || task.completed) return;

      checkMidnightRollover();

      task.completed = true;

     

      const todayStr = formatDate(new Date());

      state.completionHistory[todayStr] = (state.completionHistory[todayStr] || 0) + 1;

     

      if (auto || !state.audioEnabled) {

         playSynthSound('congrats');

         confetti({ particleCount: 100, spread: 70, origin: { y: 0.6 } });

      if (autoRemove) {

          state.tasks = state.tasks.filter(t => t.id !== id);

          saveState();

          renderTasks();

          updateDashboard();

        }, auto ? 0 : 800);

      }

      saveState();

      renderTasks();

      updateDashboard();

    }

    let audioCtx = null;

    function playSynthSound(type) {

      if (!state.audioEnabled) return;

      try {

        if (!audioCtx) {

          audioCtx = new (window.AudioContext || window.webkitAudioContext)();

        }

        if (audioCtx.state === 'suspended') {

          audioCtx.resume();

        }

        const osc = audioCtx.createOscillator();

        const gainNode = audioCtx.createGain();

        osc.connect(gainNode);

        gainNode.connect(audioCtx.destination);

        if (type === 'tick') {

          osc.type = 'triangle';

          osc.frequency.setValueAtTime(300, audioCtx.currentTime);

          osc.frequency.exponentialRampToValueAtTime(80, audioCtx.currentTime + 0.08);

          gainNode.gain.setValueAtTime(0.12, audioCtx.currentTime);

          gainNode.gain.linearRampToValueAtTime(0, audioCtx.currentTime + 0.08);

          osc.start();

          osc.stop(audioCtx.currentTime + 0.08);

        } else if (type === 'congrats') {

          const now = audioCtx.currentTime;

          osc.type = 'sine';

          osc.frequency.setValueAtTime(260, now);

          osc.frequency.exponentialRampToValueAtTime(520, now + 0.15);

          osc.frequency.exponentialRampToValueAtTime(780, now + 0.3);

          gainNode.gain.setValueAtTime(0.15, now);

          gainNode.gain.exponentialRampToValueAtTime(0.01, now + 0.35);

          osc.start();

          osc.stop(now + 0.35);

          const osc2 = audioCtx.createOscillator();

          const gainNode2 = audioCtx.createGain();

          osc2.type = 'triangle';

          osc2.connect(gainNode2);

          gainNode2.connect(audioCtx.destination);

          osc2.frequency.setValueAtTime(330, now);

          osc2.frequency.exponentialRampToValueAtTime(660, now + 0.2);

          osc2.frequency.exponentialRampToValueAtTime(990, now + 0.35);

          gainNode2.gain.setValueAtTime(0.08, now);

          gainNode2.gain.exponentialRampToValueAtTime(0.01, now + 0.45);

          osc2.start();

          osc2.stop(now + 0.45);

        } else if (type === 'complete') {

          const now = audioCtx.currentTime;

          osc.type = 'sine';

          osc.frequency.setValueAtTime(580, now);

          osc.frequency.setValueAtTime(880, now + 0.1);

          gainNode.gain.setValueAtTime(0.15, now);

          gainNode.gain.exponentialRampToValueAtTime(0.01, now + 0.3);

          osc.start();

          osc.stop(now + 0.3);

        } else if (type === 'click') {

          osc.type = 'sine';

          osc.frequency.setValueAtTime(150, audioCtx.currentTime);

          gainNode.gain.setValueAtTime(0.05, audioCtx.currentTime);

          gainNode.gain.linearRampToValueAtTime(0, audioCtx.currentTime + 0.05);

          osc.start();

          osc.stop(audioCtx.currentTime + 0.05);

        }

      } catch (e) {

        console.warn('Audio synthesis block failed.', e);

      }

    }

    function startRandomPicker() {

      checkMidnightRollover();

      const eligibleTasks = state.tasks.filter(t => !t.completed);

     

      if (eligibleTasks.length === 0) {

        showCustomError("Please add or activate at least one pending task to cycle pick!");

        return;

      }

      if (isRolling) return;

      isRolling = true;

      const chooseBtn = document.getElementById('chooseBtn');

      const pickerBox = document.getElementById('pickerBox');

      const pickerText = document.getElementById('pickerText');

      chooseBtn.disabled = true;

      chooseBtn.className = "w-full py-4 px-6 bg-slate-800 text-slate-500 font-bold rounded-xl flex items-center justify-center gap-3 text-lg cursor-not-allowed";

      chooseBtn.querySelector('span').textContent = "DECIDING DESTINY...";

      pickerBox.classList.add('animating-picker');

      // Randomize the total duration (between 2.5 and 4.0 seconds) so the final target is truly unpredictable

      let duration = 2500 + Math.floor(Math.random() * 1500);

      let startTime = Date.now();

     

      // Start cycling from a completely random index instead of zero

      let currentIndex = Math.floor(Math.random() * eligibleTasks.length);

      function roll() {

        const elapsed = Date.now() - startTime;

       

        if (elapsed >= duration) {

          // Time is up, pick the current item!

          finishPicking(eligibleTasks[currentIndex]);

        } else {

          // Cycle to the next item visually

          currentIndex = (currentIndex + 1) % eligibleTasks.length;

          pickerText.innerHTML = `

            <span class="text-blue-400 font-bold text-base block sm:text-lg animate-pulse tracking-wide truncate">

              🔮 ${eligibleTasks[currentIndex].text}

            </span>

          `;

          playSynthSound('tick');

         

          // Exponential deceleration curve

          const progress = elapsed / duration;

          const delay = 50 + Math.pow(progress, 3) * 350;

          setTimeout(roll, delay);

        }

      }

      roll();

    }

    function handleModalAction(action) {

      if (!state.currentPickId) {

        closeCongratsModal();

        return;

      }

      if (action === 'complete') {

        markTaskComplete(state.currentPickId);

      } else if (action === 'keep') {

        showFeedbackToast("Kept in backlog, keep striving!");

      }

      // Any explicit action on the modal cancels the timer

      clearActiveTimer();

      state.currentPickId = null;

      closeCongratsModal();

    }

    function updateDashboard() {

      const todoCount = state.tasks.filter(t => !t.completed).length;

      document.getElementById('statTodo').textContent = todoCount;

      const todayStr = formatDate(new Date());

      document.getElementById('statCompleted').textContent = state.completionHistory[todayStr] || 0;

      calculateStreak();

      renderGithubGrid();

    }

    function calculateStreak() {

      let streak = 0;

      const today = new Date();

      let checkDate = new Date(today);

      while (true) {

        const dateStr = formatDate(checkDate);

        const count = state.completionHistory[dateStr] || 0;

        if (count > 0) {

          streak++;

          checkDate.setDate(checkDate.getDate() - 1);

        } else {

          if (formatDate(today) === dateStr) {

            checkDate.setDate(checkDate.getDate() - 1);

            continue;

          }

          break;

        }

      }

      state.streak = streak;

      document.getElementById('statStreak').textContent = `${streak}d`;

    }
